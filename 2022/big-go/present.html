spaghetti-cutter - General Configuration

{
    tool: ["x/*"]

    db: ["db/*", "/"]

    // not necessary as any 'main' package is a god package by default
    god: ["cmd/*"]

    // rarely useful but you can forbid default god package 'main'
    noGod: true

    // default size (in minimal lines of code):
    size: 2048
}
50

spaghetti-cutter - Allow Configuration

{
    tool: [ ... ]
    db: [ ... ]
    size: 1024

    allowAdditionally: {
        // read: allow "superdomain" to additionally import ...

        "superdomain": ["domainx/subdomain1", "domainy/subdomain2"]
        "x/supertool": ["x/tool1", "db/*", "domainx"]
    }
}
51

spaghetti-cutter - Restrict Configuration

{
    tool: [ ... ]
    db: [ ... ]
    size: 1024

    allowAdditionally: { ... }

    allowOnlyIn: {
        // read: allow "domainx/subdomain1" only to be imported in ...
        "domainx/subdomain1": ["domainx*", "superdomain"]

        // the only config key that can be used for external packages
        "github.com/lib/pq": ["main"]
        "github.com/jmoiron/sqlx": ["db/**"]
    }
}
